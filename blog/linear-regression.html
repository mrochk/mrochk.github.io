# We need to define a way to tell our model how bad is its prediction
# we use the MSE

def mse(preds, target):
    sum = 0
    for i in range(len(preds)):
        sum += (preds[i] - target[i])**2
    return sum / len(preds)

# Now we need to calculate the gradients to update our model:
# da = -2/N * Sum(i=1 -> N){x[i]^2 * (y[i] - (ax[i]^2 + bx[i] + c))}
# db = -2/N * Sum(i=1 -> N){x[i] * (y[i] - (ax[i]^2 + bx[i] + c))}
# dc = -2/N * Sum(i=1 -> N){y[i] * (y[i] - (ax[i]^2 + bx[i] + c))}

def calculate_gradients(X, Y, params):
    a, b, c = params
    da = db = dc = 0
    N = len(X)
    for i in range(N):
        da += X[i]**2 * (Y[i] - (a*(X[i]**2) + b*X[i] + c))
        db += X[i] * (Y[i] - (a*(X[i]**2) + b*X[i] + c))
        dc += Y[i]**2 - Y[i]*(a*(X[i]**2) + b*X[i] + c)
    da *= -2 / N; db *= -2 / N; dc *= -2 / N
    return da, db, dc
